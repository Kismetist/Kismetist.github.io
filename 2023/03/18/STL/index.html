单调栈

#include <bits/stdc++.h>
using namespace std;
#define pii pair<int,int>
int a[100010],L[100010],R[100010];
int n;

stack<pii> s;
void Find(int Ans[])//向左找到第一个比自己小的
{
    while(!s.empty())s.pop();
    for(int i=1;i<=n;i++)
    {
        while(!s.empty()&&s.top().first>=a[i])
        s.pop();//对于每一个点，向左找到第一个比自己矮的
        if(s.empty())//没有比自己矮的
        Ans[i] = i;
        else
        Ans[i] = i - s.top().second;//有效距离，比他矮的是用不到的
        s.push(pii(a[i],i));//加入作为被比较者
    }
}

int main()
{
    cin>>n;
    for(int i = 1;i<=n;i++)
    cin>>a[i];
    Find(L);
    reverse(a+1,a+1+n);
    Find(R);
    reverse(a+1,a+1+n);
    reverse(R+1,R+1+n);
    long long Ans = 0;
    for(int i = 1;i<=n;i++)
    {
        long long temp = a[i]*1ll*(R[i]+L[i]-1);
        Ans = max(temp,Ans);
    }
    cout<<Ans<<endl;
    //system("pause");
    return 0;
}

迪杰斯特拉最短路

#include <bits/stdc++.h>
using namespace std;
const int maxm = 1e6+7;
const int maxn = 1e6+7;
const int INF = 1e8+7;
int n,m,T;
struct Edge{
    int v;
    int w;
    int t;
    int nxt;
}E[maxm];

int Head[maxn],tot;

void init()
{
    for(int i=1;i<=n;i++)Head[i] = -1;
    tot = 0;
}

void addEdge(int u,int v,int w,int t)
{
    E[tot].v = v;
    E[tot].w = w;
    E[tot].t = t;
    E[tot].nxt = Head[u];
    Head[u] = tot;
    tot++;
}


int dis[maxn],vis[maxn];
bool Dijkstra(int C)
{
    for(int i = 1;i<=n;i++)
    {
        dis[i] = INF;
        vis[i] = 0;
    }
    dis[1] = 0;
    priority_queue<pair<int,int>> q;
    q.push(make_pair(0,1));
    while(!q.empty())
    {
        int u = q.top().second;
        q.pop();
        if(vis[u])continue;
        vis[u] = 1;

        for(int i = Head[u];i!=-1;i=E[i].nxt)
        {
            int v = E[i].v;
            int w = E[i].w;
            int t = E[i].t;
            if((dis[v]>dis[u]+t)&&w>=C)
            {
                dis[v] = dis[u]+t;
                q.push(make_pair(-dis[v],v));
            }
        }
    }
    if(dis[n]>T)
    return false;
    else
    return true;
}

int main()
{
    int X;
    cin>>X;
    while(X!=0)
    {
        X--;
        cin>>n>>m>>T;
        init();
        for(int i=1;i<=m;i++)
        {
            int u,v,w,t;
            cin>>u>>v>>w>>t;
            addEdge(u,v,w,t);
            addEdge(v,u,w,t);
        }
        int l = 0;
        int r = INF;
        while(r-l!=1)
        {
            int mid = (r+l)/2;
            bool deter =  Dijkstra(mid);
            if(deter)
            l = mid;
            else
            r = mid;
        }
        cout<<l<<endl;
    }
    //system("pause");
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
const int maxm = 1e7;
const int maxn = 1e7;
const int INF = 1e8;
int n,m;
int dis[maxn];
int vis[maxn];
struct EdgeData{
    int v;
    int w;
    int nxt;
}E[maxm];

int head[maxn],tot;

void init()
{
    for(int i=0;i<=n;i++)
    head[i] = -1;
    tot=0;
}

void addEdge(int u,int v,int w)
{
    E[tot].v = v;
    E[tot].w = w;
    E[tot].nxt = head[u];
    head[u] = tot;
    tot++;
}

int cnt[maxn];//记录从源点到i最短路所走的边数
bool Spfa(int s)
{
    for(int i =0;i<=n;i++)dis[i] = INF,vis[i] = 0,cnt[i] = 0;
    queue<int> q;
    dis[s] = 0;
    q.push(s);
    vis[s]=1;//表示当前节点是否在队列中

    while(!q.empty())
    {
        int u = q.front();
        q.pop();
        vis[u] = 0;
        //u点出了队列，那么一个点就可以多次入队

        for(int i = head[u];i!=-1;i =E[i].nxt)
        {
            int v = E[i].v;
            int t = E[i].w;
            if(dis[v]>dis[u]+t)
            {
                dis[v] = dis[u]+t;
                cnt[v] = cnt[u]+1;
                if(cnt[v]>=n)
                return false;
                if(!vis[v])
                q.push(v);
            }
        }
    }
    return true;
}

int main()
{
    cin>>n>>m;
    init();
    for(int i = 1;i<=m;i++)
    {
        int k,a,b,c;
        cin>>k>>a>>b>>c;
        a=a-1;
        if(k==1)
            addEdge(a,b,c);
        if(k==2)
            addEdge(b,a,-c);
        if(k==3)
            addEdge(a,b,c-1);
        if(k==4)
            addEdge(b,a,-c-1);
        if(k==5)
        {
            addEdge(a,b,c);
            addEdge(b,a,-c);
        }
    }
    for(int i = 0;i<=n-1;i++)
    {
        addEdge(i,i+1,1);
        addEdge(i+1,i,0);
    }
    bool res = Spfa(0);
    if(res)
    cout<<dis[n]<<endl;
    else
    cout<<"impossible\n"<<endl;
    //system("pause");
    return 0;
}
